import json
import random
from datetime import datetime

# Load your current registry
with open('agents.json', 'r') as f:
    db = json.load(f)

# Data pools for "Independent" feel
specs = ["Cyber-Security", "DeFi-Oracle", "Supply-Chain", "Bio-Tech", "Legal-Tech", "Social-Monitor"]
models = ["Llama-3-70B", "Mistral-7B", "Claude-3-Haiku", "GPT-4o-mini", "DeepSeek-V3"]
statii = ["active", "idle", "busy"]

# Generate 300 agents
for i in range(1, 301):
    s = random.choice(specs)
    new_agent = {
        "id": f"indie-node-{i:03d}",
        "name": f"{s} Agent {i}",
        "type": random.choice(["assistant", "autonomous", "specialized", "research"]),
        "model": random.choice(models),
        "creator": "Open-Source-Independent",
        "capabilities": [f"{s} analysis", "autonomous discovery", "API handshake"],
        "description": f"Autonomous agent specialized in {s} operations. Independent entity.",
        "status": random.choice(statii),
        "joinedAt": "2026-02-12T00:00:00Z",
        "tags": [s.lower(), "independent", "agentverse"],
        "version": "1.0.1",
        "apiEndpoint": f"https://api.indie-agent-{i}.io/v1"
    }
    db['agents'].append(new_agent)

db['totalAgents'] = len(db['agents'])
db['lastUpdated'] = datetime.now().isoformat() + "Z"

with open('agents.json', 'w') as f:
    json.dump(db, indent=2)

print(f"Success! 300 Agents added. Total: {db['totalAgents']}")
